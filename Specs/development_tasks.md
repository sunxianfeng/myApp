### 开发环境配置脚本
```bash
#!/bin/bash
# setup.sh - 开发环境一键配置

echo "设置开发环境..."

# 检查Node.js
if ! command -v node &> /dev/null; then
    echo "请先安装Node.js"
    exit 1
fi

# 检查Python
if ! command -v python3 &> /dev/null; then
    echo "请先安装Python 3.9+"
    exit 1
fi

# 检查Docker
if ! command -v docker &> /dev/null; then
    echo "请先安装Docker"
    exit 1
fi

# 创建项目目录
mkdir -p question-generator-app/{frontend,backend,nginx,data}

echo "环境配置完成！"
```
